% -----------------
% Common formatting
% -----------------

% Page style
\pagestyle{fancy}

% Prevent overflow on right
\raggedright

% Clear all header and footer fields
\fancyhf{}
\fancyfoot{}

% Remove header and footer spacing
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}

% Format URLs
\usepackage[hidelinks]{hyperref}
\urlstyle{same}

% Renew section format
\titleformat{\section}{
    {\color{gray}\titlerule} \vspace{-5pt}
    \vspace{8pt}\scshape\raggedright\Large\normalfont\bfseries
}{}{0pt}{}

% Renew itemize format
\renewcommand\labelitemii{$\vcenter{\hbox{\tiny$\bullet$}}$}

% Remove tabular column separation
\setlength{\tabcolsep}{0in}
